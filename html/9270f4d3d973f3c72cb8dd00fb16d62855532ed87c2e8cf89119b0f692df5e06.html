JSON_Number

type JSON_Number (Kind : JSON_Number_Kind := None) is record
   case Kind is
      when None =>
         null;
      when others =>
         String_Value : VSS.Strings.Virtual_String;
         case Kind is
            when None | Out_Of_Range =>
               null;
            when JSON_Integer =>
               Integer_Value : Interfaces.Integer_64;
            when JSON_Float =>
               Float_Value   : Interfaces.IEEE_Float_64;
         end case;
   end case;
end record;
